def establish_connection
  cache_dir = File.join(Rails.root, '/tmp', 'cache')

  Faraday.new(url: base_url) do |conn|
    conn.request :retry, request_retry_opts
    conn.request :url_encoded
    conn.response :json, content_type: /\bjson$/
    conn.response :xml,  content_type: /\bxml$/
    conn.response :caching, :ignore_params => %w[access_token] do  
      ActiveSupport::Cache::FileStore.new cache_dir, :namespace => 'ils_holds',
        :expires_in => 3600  # one hour
    end
    conn.adapter :typhoeus
  end

end

def process_response(resp)
  resp_con_type = response_content_type(resp)
  if resp_con_type =~ /#{response_format}/
    resp.body.to_ostruct
    # resp.body
  else
    fail Error, "Response format does not match expected type (Expected: #{response_format}, Received: #{resp_con_type})"
  end
end

def request_retry_opts
  {
    max: max_retries,
    interval: 0.05,
    interval_randomness: 1,
    backoff_factor: 2
  }
end
